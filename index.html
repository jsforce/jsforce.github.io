Connect to Salesforce

Develop Salesforce apps with JavaScript,
whether it's running on server or web browser.

Download View on GitHub

  • Node.js
  • Web Browser (Visualforce)
  • Web Browser (Outer Website)
  • Command Line Interface (CLI)

Setup

$ npm install jsforce

Run

var jsforce = require('jsforce');
var conn = new jsforce.Connection();
conn.login('username@domain.com', 'password', function(err, res) {
  if (err) { return handleError(err); }
  handleResult(res);
});

Run

<apex:page>
  <apex:includeScript value="{!URLFOR($Resources.JSforce)}" />
  <script>
var conn = new jsforce.Connection({ accessToken: '{!$API.Session_Id}' });
conn.query('SELECT Id, Name FROM Account', function(err, res) {
  if (err) { return handleError(err); }
  handleResult(res);
});
  </script>
</apex:page>

Run

<script src="/path/to/jsforce.js"></script>
<script>
jsforce.browser.init({ clientId: '[ your Salesforce OAuth2 ClientID is here ]' });
jsforce.browser.on('connect', function(conn) {
  conn.query('SELECT Id, Name FROM Account', function(err, res) {
    if (err) { return handleError(err); }
    handleResult(res);
  });
});
</script>
<button onclick="javascript:jsforce.browser.login();">Login</button>
